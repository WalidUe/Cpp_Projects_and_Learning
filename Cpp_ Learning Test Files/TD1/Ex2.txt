int n_premier(int nb){
	int diviseurs = 0;
	int rest=0;
	for (int i = 2; i <= nb; i++){
		if (((nb % i) == 0 )&& ((nb % 1) == 0)){
			diviseurs++;
		}
	}
	return diviseurs;
}
